  var point = {x:0, y:0};
  if (event.targetTouches) {
    point.x = event.targetTouches[0].clientX;
    point.y = event.targetTouches[0].clientY;

    point.x = event.clientX;
    point.y = event.clientY;

function onPointerMove (event) {
